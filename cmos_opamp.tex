\draw (1,1.5) to[nmos, l=$M_5$] (1,0) node[ground]{};

\draw (2,1.5) -- (1,1.5);

\draw (0,4) -- (0,3.5);
\draw (2,4) -- (2,3.5);

% Second stage gain transistor M6
\draw (2,4) -- (4,4) to[pmos, l_=$M_6$] (4,2.5);

\draw (4,2.5) to[nmos, l_=$M_7$] (4,0) node[ground]{};

% Output
\draw (4,2.5) -- (5,2.5) node[right]{$V_{out}$};
